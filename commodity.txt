  <script>
    const API = '';
    const symbol = decodeURIComponent(window.location.pathname.replace(/\/(stock|commodity)\//, ''));
    let watchlist = JSON.parse(localStorage.getItem('fp_watchlist') || '[]');
    let mainChartInst = null;
    let techChartInst = null;
    let chartType = 'line';
    let lastHistData = null;
    let prevPrice = null;

    const fmt = (n, d = 2) => {
      if (n == null || isNaN(n)) return '--';
      return Number(n).toLocaleString('zh-TW', { minimumFractionDigits: d, maximumFractionDigits: d });
    };
    const fmtVol = n => {
      if (!n || isNaN(n)) return '--';
      if (n >= 1e9) return (n / 1e9).toFixed(2) + 'B';
      if (n >= 1e8) return (n / 1e8).toFixed(1) + '億';
      if (n >= 1e4) return (n / 1e4).toFixed(0) + '萬';
      return n.toLocaleString();
    };
    const cls = v => v > 0 ? 'up' : v < 0 ? 'down' : '';
    const arrow = v => v > 0 ? '▲' : v < 0 ? '▼' : '–';

    // ── TABS ──
    document.querySelectorAll('#mainTabs .tab-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        document.querySelectorAll('#mainTabs .tab-btn').forEach(b => b.classList.remove('active'));
        document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
        btn.classList.add('active');
        document.getElementById('tab-' + btn.dataset.tab).classList.add('active');
        if (btn.dataset.tab === 'flows' && !flowsLoaded) loadFlows();
        if (btn.dataset.tab === 'tech') loadTechChartDefault();
      });
    });

    // ── WATCHLIST ──
    function updateWlBtn() {
      const btn = document.getElementById('wlBtn');
      const inWl = watchlist.includes(symbol);
      btn.textContent = inWl ? '★ 已加入自選' : '☆ 加入自選股';
      btn.classList.toggle('active', inWl);
    }
    function toggleWatchlist() {
      const idx = watchlist.indexOf(symbol);
      if (idx > -1) watchlist.splice(idx, 1);
      else watchlist.push(symbol);
      localStorage.setItem('fp_watchlist', JSON.stringify(watchlist));
      updateWlBtn();
    }
    updateWlBtn();

    // ── LOAD DETAIL ──
    async function loadDetail() {
      try {
        const r = await fetch(`${API}/api/stock/${encodeURIComponent(symbol)}/detail`);
        const j = await r.json();
        const d = j.data || {};

        const priceEl = document.getElementById('heroPrice');
        const newPrice = d.price;
        if (prevPrice !== null && prevPrice !== newPrice) {
          const flashClass = newPrice > prevPrice ? 'flash-up' : 'flash-down';
          priceEl.classList.add(flashClass);
          setTimeout(() => priceEl.classList.remove(flashClass), 700);
        }
        prevPrice = newPrice;

        document.title = `${d.symbol} ${fmt(d.price)} — FinPulse`;
        document.getElementById('heroSym').textContent = d.symbol || symbol;
        document.getElementById('heroName').textContent = d.name || symbol;
        priceEl.textContent = fmt(d.price);
        const chgEl = document.getElementById('heroChange');
        const chgPct = d.change_pct || 0;
        const chg = d.change || 0;
        chgEl.className = `hero-change ${cls(chgPct)}`;
        chgEl.textContent = `${arrow(chgPct)} ${fmt(Math.abs(chg))} (${Math.abs(chgPct).toFixed(2)}%)`;

    function updateTechIndicators(tech) {
      const rsi = tech.rsi;
      if (rsi !== null && rsi !== undefined) {
        document.getElementById('ti-rsi').textContent = fmt(rsi, 1);
        const rsiSig = document.getElementById('ti-rsi-sig');
        if (rsi < 30) { rsiSig.textContent = '超賣 / 買入訊號'; rsiSig.className = 'tech-signal buy'; }
        else if (rsi > 70) { rsiSig.textContent = '超買 / 賣出訊號'; rsiSig.className = 'tech-signal sell'; }
        else { rsiSig.textContent = '中性區間'; rsiSig.className = 'tech-signal neutral'; }
      }
      if (tech.macd !== null && tech.macd !== undefined) {
        document.getElementById('ti-macd').textContent = fmt(tech.macd, 4);
        const macdSig = document.getElementById('ti-macd-sig');
        if (tech.macd > 0) { macdSig.textContent = '多頭'; macdSig.className = 'tech-signal buy'; }
        else { macdSig.textContent = '空頭'; macdSig.className = 'tech-signal sell'; }
      }
      if (tech.signal) document.getElementById('ti-signal').textContent = fmt(tech.signal, 4);
      if (tech.bb_upper) document.getElementById('ti-bb-up').textContent = fmt(tech.bb_upper);
      if (tech.bb_middle) document.getElementById('ti-bb-mid').textContent = fmt(tech.bb_middle);
      if (tech.bb_lower) document.getElementById('ti-bb-low').textContent = fmt(tech.bb_lower);
    }

    function renderFinancials(d) {
      const fmtBig = n => {
        if (!n || n === 0) return '--';
        if (Math.abs(n) >= 1e12) return (n / 1e12).toFixed(2) + ' T';
        if (Math.abs(n) >= 1e9) return (n / 1e9).toFixed(2) + ' B';
        if (Math.abs(n) >= 1e6) return (n / 1e6).toFixed(2) + ' M';
        return fmt(n, 0);
      };
      const cards = [
        { icon: '💰', label: '年營收', value: fmtBig(d.revenue) },
        { icon: '📈', label: '毛利', value: fmtBig(d.gross_profit) },
        { icon: '💵', label: '淨利', value: fmtBig(d.net_income) },
        { icon: '🔄', label: '營業現金流', value: fmtBig(d.operating_cashflow) },
        { icon: '📊', label: '本益比 (P/E)', value: d.pe_ratio ? fmt(d.pe_ratio, 1) : '--' },
        { icon: '📉', label: '股價淨值比', value: d.pb_ratio ? fmt(d.pb_ratio, 2) : '--' },
        { icon: '📋', label: 'EPS', value: d.eps ? fmt(d.eps, 2) : '--' },
        { icon: '💸', label: '殖利率', value: d.dividend_yield ? fmt(d.dividend_yield, 2) + '%' : '--' },
      ];
      document.getElementById('finGrid').innerHTML = cards.map(c => `
        <div class="fin-card">
          <div class="fin-card-label"><span class="icon">${c.icon}</span>${c.label}</div>
          <div class="fin-card-value">${c.value}</div>
        </div>`).join('');

      const ratios = [
        ['負債/淨值比', d.debt_to_equity ? fmt(d.debt_to_equity, 2) : '--', '< 2 通常較安全'],
        ['流動比率', d.current_ratio ? fmt(d.current_ratio, 2) : '--', '> 1.5 表示短期償債能力佳'],
        ['股東權益報酬率', d.return_on_equity ? (d.return_on_equity * 100).toFixed(2) + '%' : '--', '越高表示獲利能力越強'],
        ['淨利率', d.profit_margin ? (d.profit_margin * 100).toFixed(2) + '%' : '--', '營業收入中有多少轉為淨利'],
        ['本益比', d.pe_ratio ? fmt(d.pe_ratio, 1) : '--', '股價/每股盈餘，越低越便宜'],
        ['EPS', d.eps ? fmt(d.eps, 2) : '--', '每股盈餘'],
      ];
      document.getElementById('ratioBody').innerHTML = ratios.map(([label, val, desc]) => `
        <tr>
          <td style="text-align:left;color:var(--text2)">${label}</td>
          <td style="font-family:'JetBrains Mono',monospace;font-weight:600">${val}</td>
          <td style="text-align:left;color:var(--muted);font-size:11px">${desc}</td>
        </tr>`).join('');
    }

    async function loadHistory(period = '1mo', interval = '1d') {
      const loading = document.getElementById('chartLoading');
      loading.style.display = 'flex';
      try {
        const r = await fetch(`${API}/api/stock/${encodeURIComponent(symbol)}/history?period=${period}&interval=${interval}`);
        const j = await r.json();
        lastHistData = j.data || [];
        renderChart(lastHistData, chartType);
      } catch (e) { console.error('History error:', e); }
      finally { loading.style.display = 'none'; }
    }

    function setChartType(type) {
      chartType = type;
      document.querySelectorAll('.ct-btn').forEach(b => b.classList.remove('active'));
      document.getElementById(`ct${type.charAt(0).toUpperCase() + type.slice(1)}`).classList.add('active');
      if (lastHistData) renderChart(lastHistData, chartType);
    }

    function renderChart(data, type) {
      const canvas = document.getElementById('mainChart');
      const ctx = canvas.getContext('2d');
      if (mainChartInst) { mainChartInst.destroy(); mainChartInst = null; }
      if (!data.length) return;

      const labels = data.map(d => d.date);
      const closes = data.map(d => d.close);
      const first = closes[0], last = closes[closes.length - 1];
      const color = last >= first ? '#00e676' : '#ff3d5a';

      if (type === 'candle') {
        // Candlestick using simple bar chart simulation
        const opens = data.map(d => d.open);
        const highs = data.map(d => d.high);
        const lows = data.map(d => d.low);
        // Render as grouped bar chart
        mainChartInst = new Chart(ctx, {
          type: 'bar',
          data: {
            labels,
            datasets: [
              {
                label: '高-低',
                data: data.map(d => [d.low, d.high]),
                backgroundColor: data.map(d => d.close >= d.open ? 'rgba(0,230,118,0.3)' : 'rgba(255,61,90,0.3)'),
                borderColor: data.map(d => d.close >= d.open ? '#00e676' : '#ff3d5a'),
                borderWidth: 1,
                borderSkipped: false,
              },
              {
                label: '開-收',
                data: data.map(d => [Math.min(d.open, d.close), Math.max(d.open, d.close)]),
                backgroundColor: data.map(d => d.close >= d.open ? '#00e676' : '#ff3d5a'),
                borderColor: data.map(d => d.close >= d.open ? '#00e676' : '#ff3d5a'),
                borderWidth: 1,
                borderSkipped: false,
              },
            ]
          },
          options: {
            responsive: true, maintainAspectRatio: false,
            plugins: { legend: { display: false }, tooltip: {
              callbacks: {
                title: (items) => labels[items[0].dataIndex],
                label: (item) => {
                  const d = data[item.dataIndex];
                  return [`開: ${fmt(d.open)}`, `高: ${fmt(d.high)}`, `低: ${fmt(d.low)}`, `收: ${fmt(d.close)}`];
                }
              }
            }},
            scales: {
              x: { grid: { color: 'rgba(33,39,58,0.4)' }, ticks: { color: '#5c6a85', maxTicksLimit: 10, font: { size: 10 } } },
              y: { position: 'right', grid: { color: 'rgba(33,39,58,0.4)' }, ticks: { color: '#5c6a85', font: { size: 10 } } }
            },
            interaction: { mode: 'index', intersect: false }
          }
        });
        return;
      }

      // Line or Area
      const grad = ctx.createLinearGradient(0, 0, 0, 380);
      if (type === 'area') {
        grad.addColorStop(0, color + '30');
        grad.addColorStop(1, color + '00');
      }

      mainChartInst = new Chart(ctx, {
        type: 'line',
        data: {
          labels,
          datasets: [{
            label: 'Close',
            data: closes,
            borderColor: color,
            backgroundColor: type === 'area' ? grad : 'transparent',
            fill: type === 'area',
            tension: 0.2,
            pointRadius: 0,
            pointHoverRadius: 5,
            pointHoverBackgroundColor: color,
            borderWidth: 2,
          }]
        },
        options: {
          responsive: true, maintainAspectRatio: false,
          plugins: {
            legend: { display: false },
            tooltip: {
              backgroundColor: 'rgba(13,17,23,0.95)',
              borderColor: '#21273a', borderWidth: 1,
              titleColor: '#a8b3cc', bodyColor: '#e6eaf4',
              padding: 10, cornerRadius: 8,
              callbacks: {
                label: (item) => ` ${item.dataset.label}: ${fmt(item.raw)}`
              }
            }
          },
          scales: {
            x: { grid: { color: 'rgba(33,39,58,0.4)' }, ticks: { color: '#5c6a85', maxTicksLimit: 10, font: { size: 10 } } },
            y: { position: 'right', grid: { color: 'rgba(33,39,58,0.4)' }, ticks: { color: '#5c6a85', font: { size: 10 } } }
          },
          interaction: { mode: 'index', intersect: false }
        }
      });
    }

    async function loadTechChart(btn) {
      if (techPeriodBtn) techPeriodBtn.classList.remove('active');
      btn.classList.add('active');
      techPeriodBtn = btn;
      const period = btn.dataset.period2 || '1mo';
      const interval = btn.dataset.interval2 || '1d';
      try {
        const r = await fetch(`${API}/api/stock/${encodeURIComponent(symbol)}/history?period=${period}&interval=${interval}`);
        const j = await r.json();

        const tech = j.tech || {};
        updateTechIndicators(tech);
        renderTechChart(data);
      } catch (e) {}
    }

    function renderTechChart(data) {
      const ctx = document.getElementById('techChart').getContext('2d');
      if (techChartInst) { techChartInst.destroy(); techChartInst = null; }
      if (!data.length) return;

      const labels = data.map(d => d.date);
      const closes = data.map(d => d.close);

      const sma20 = closes.map((v, i, arr) => {
        if (i < 19) return null;
        return arr.slice(i - 19, i + 1).reduce((a, b) => a + b) / 20;
      });
      const sma5 = closes.map((v, i, arr) => {
        if (i < 4) return null;
        return arr.slice(i - 4, i + 1).reduce((a, b) => a + b) / 5;
      });

      techChartInst = new Chart(ctx, {
        type: 'line',
        data: {
          labels,
          datasets: [
            { label: '收盤價', data: closes, borderColor: '#4f8ef7', yAxisID: 'y', tension: 0.2, pointRadius: 0, borderWidth: 2 },
            { label: 'SMA(5)', data: sma5, borderColor: '#ffc542', borderDash: [4, 4], yAxisID: 'y', tension: 0.2, pointRadius: 0, borderWidth: 1.5 },
            { label: 'SMA(20)', data: sma20, borderColor: '#a78bfa', borderDash: [6, 3], yAxisID: 'y', tension: 0.2, pointRadius: 0, borderWidth: 1.5 },
          ]
        },
        options: {
          responsive: true, maintainAspectRatio: false,
          plugins: {
            legend: { display: true, position: 'top', labels: { color: '#a8b3cc', font: { size: 11 }, padding: 12, boxWidth: 16 } },
            tooltip: {
              backgroundColor: 'rgba(13,17,23,0.95)', borderColor: '#21273a', borderWidth: 1,
              titleColor: '#a8b3cc', bodyColor: '#e6eaf4', padding: 10, cornerRadius: 8,
            }
          },
          scales: {
            x: { grid: { color: 'rgba(33,39,58,0.3)' }, ticks: { color: '#5c6a85', maxTicksLimit: 8, font: { size: 10 } } },
            y: { position: 'right', grid: { color: 'rgba(33,39,58,0.3)' }, ticks: { color: '#5c6a85', font: { size: 10 } } }
          },
          interaction: { mode: 'index', intersect: false }
        }
      });
    }

    async function loadFlows() {
      flowsLoaded = true;
      const container = document.getElementById('flowsContent');
      try {
        const r = await fetch(`${API}/api/stock/${encodeURIComponent(symbol)}/flows`);
        const j = await r.json();

        }

            <tr>
              <th>日期</th>
              <th>外資買進(張)</th><th>外資賣出(張)</th><th>外資淨買(張)</th>
              <th>投信買進(張)</th><th>投信賣出(張)</th><th>投信淨買(張)</th>
              <th>自營商淨買(張)</th>
            </tr>
          </thead>
          <tbody>${flows.map(f => {
            const fnc = f.foreign_net > 0 ? 'flow-up' : f.foreign_net < 0 ? 'flow-down' : '';
            const tnc = f.trust_net > 0 ? 'flow-up' : f.trust_net < 0 ? 'flow-down' : '';
            const dnc = f.dealer_net > 0 ? 'flow-up' : f.dealer_net < 0 ? 'flow-down' : '';
            return `<tr>
              <td>${f.date}</td>
              <td>${f.foreign_buy?.toLocaleString() || '--'}</td>
              <td>${f.foreign_sell?.toLocaleString() || '--'}</td>
              <td class="${fnc}">${f.foreign_net >= 0 ? '+' : ''}${f.foreign_net?.toLocaleString() || '--'}</td>
              <td>${f.trust_buy?.toLocaleString() || '--'}</td>
              <td>${f.trust_sell?.toLocaleString() || '--'}</td>
              <td class="${tnc}">${f.trust_net >= 0 ? '+' : ''}${f.trust_net?.toLocaleString() || '--'}</td>
              <td class="${dnc}">${f.dealer_net >= 0 ? '+' : ''}${f.dealer_net?.toLocaleString() || '--'}</td>
            </tr>`;
          }).join('')}</tbody>
        </table>`;
      } catch (e) {
        container.innerHTML = '<div style="padding:30px;color:var(--muted);text-align:center">無法載入法人資料</div>';
      }
    }
